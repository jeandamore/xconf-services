              post '/' do
                if @account_service.get(params[:email.to_s]).code == 200
                  status 422
                else
                  status @account_service.post(params[:email.to_s]).code

Choreography

              post '/' do
                @q.post(Configuration.value(:queue), params[:email.to_s])
                status 201
              end
